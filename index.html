export default { async fetch(request, env) { const corsHeaders = { 'Access-Control-Allow-Origin': 'https://atlas-anesthesia.github.io', 'Access-Control-Allow-Methods': 'GET, POST, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type', }; if (request.method === 'OPTIONS') { return new Response(null, { headers: corsHeaders }); } try { const { email } = await request.json(); if (!email) { return new Response(JSON.stringify({ error: 'Email required' }), { status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } // Search Stripe for customers with this email const customerRes = await fetch( `https://api.stripe.com/v1/customers?email=${encodeURIComponent(email)}&limit=5`, { headers: { 'Authorization': `Bearer ${env.STRIPE_SECRET_KEY}`, } } ); const customerData = await customerRes.json(); if (!customerData.data || customerData.data.length === 0) { return new Response(JSON.stringify({ status: 'not_found', message: 'No customer found with this email' }), { headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } const customerId = customerData.data[0].id; const customerName = customerData.data[0].name || ''; // Search for payment intents for this customer const paymentsRes = await fetch( `https://api.stripe.com/v1/payment_intents?customer=${customerId}&limit=10`, { headers: { 'Authorization': `Bearer ${env.STRIPE_SECRET_KEY}`, } } ); const paymentsData = await paymentsRes.json(); const payments = paymentsData.data || []; // Find succeeded payments const succeeded = payments.filter(p => p.status === 'succeeded'); if (succeeded.length === 0) { return new Response(JSON.stringify({ status: 'unpaid', message: 'No completed payments found', customerName }), { headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } // Return most recent succeeded payment const latest = succeeded[0]; return new Response(JSON.stringify({ status: 'paid', amount: (latest.amount / 100).toFixed(2), currency: latest.currency.toUpperCase(), date: new Date(latest.created * 1000).toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' }), customerName, paymentId: latest.id }), { headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } catch (err) { return new Response(JSON.stringify({ error: 'Server error', detail: err.message }), { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } } };
